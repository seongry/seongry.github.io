function solution(name) {
  let answer = 0,
    start = "A".charCodeAt(0),
    end = "Z".charCodeAt(0),
    index = 0,
    reverse = false,
    init = ""

  for (let i = 0; i < name.length; i++) {
    init += "A"
  }

  while (init !== name) {
    let left = 0,
      right = 0
    if (init[index] !== name[index]) {
      answer +=
        name[index].charCodeAt(0) - start < end - name[index].charCodeAt(0)
          ? name[index].charCodeAt(0) - start
          : end - name[index].charCodeAt(0) + 1
      init = init.substring(0, index) + name[index] + init.substring(index + 1)
    }

    if (init !== name) {
      if (!reverse) {
        for (let i = index + 1; i < name.length; i++) {
          left += 1
          if (name[i] !== "A") {
            break
          }
        }

        for (let i = name.length - 1; i > index; i--) {
          right += 1
          if (name[i] !== "A") {
            break
          }
        }

        if (right < left) {
          reverse = true
        }
      }

      if (reverse && !index) {
        index = name.length - 1
      } else {
        reverse ? index-- : index++
      }

      answer += 1
    }
  }
  return answer
}
